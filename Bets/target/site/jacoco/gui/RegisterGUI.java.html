								validText.setText("Error. Try again.");

	private boolean checkEmail(String eMail){
		boolean ret = false;
		if(eMail.contains("@")) {
			String[] split = eMail.split("@");
			if(split.length == 2 && split[0].length() > 1 && split[1].contains(".")) {
				ret = true;
			}
		}
		return ret;
	}

		previous.setVisible(true);
